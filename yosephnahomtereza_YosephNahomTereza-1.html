from vpython import * #GlowScript 3.2 VPython scene.background = color.cyan scene.title = "Skydiver with Real Curved Parachute and Land" # ZOOM OUT CAMERA scene.range = 25 scene.forward = vector(0,-0.3,-1) scene.center = vector(0,6,0) # GREEN LAND ground = box(pos=vector(0,0,0), size=vector(60,0.5,60), color=color.green) # TREES tree_trunk1 = cylinder(pos=vector(6,0,2), axis=vector(0,2,0), radius=0.2, color=color.orange) tree_leaves1 = sphere(pos=vector(6,2.6,2), radius=0.9, color=color.green) tree_trunk2 = cylinder(pos=vector(-7,0,-3), axis=vector(0,2,0), radius=0.2, color=color.orange) tree_leaves2 = sphere(pos=vector(-7,2.6,-3), radius=0.9, color=color.green) # CLOUDS cloud1 = sphere(pos=vector(5,14,0), radius=1, color=color.white, opacity=0.8) cloud2 = sphere(pos=vector(-6,13,1), radius=1.2, color=color.white, opacity=0.8) # SKYDIVER BODY head = sphere(pos=vector(0,12,0), radius=0.25, color=color.yellow) body = cylinder(pos=vector(0,11.2,0), axis=vector(0,0.8,0), radius=0.08) left_arm = cylinder(pos=vector(0,11.7,0), axis=vector(-0.6,-0.3,0), radius=0.05) right_arm = cylinder(pos=vector(0,11.7,0), axis=vector(0.6,-0.3,0), radius=0.05) left_leg = cylinder(pos=vector(0,11.2,0), axis=vector(-0.3,-1,0), radius=0.06) right_leg = cylinder(pos=vector(0,11.2,0), axis=vector(0.3,-1,0), radius=0.06) # REALISTIC CURVED PARACHUTE canopy_radius = 3 canopy_height = 1.5 num_segments = 20 parachute_points = [] for i in range(num_segments + 1): angle = pi * i / num_segments y = canopy_height * sin(angle) r = canopy_radius * cos(angle) num_circle_points = 30 for j in range(num_circle_points): theta = 2 * pi * j / num_circle_points x = r * cos(theta) z = r * sin(theta) parachute_points.append(sphere(pos=vector(x, head.pos.y + y + 2, z), radius=0.1, color=color.red, visible=False)) # ROPES ropes = [] for pt in parachute_points[::num_circle_points]: rope = cylinder(pos=pt.pos, axis=head.pos - pt.pos, radius=0.03, color=color.white, visible=False) ropes.append(rope) # PHYSICS v = 0 g = 9.8 drag_small = 0.05 drag_big = 1.5 dt = 0.01 t = 0 parachute_deployed = False # LABELS vel_label = label(pos=vector(-20,18,0), text="Velocity: 0 m/s", height=16, box=False, color=color.red) acc_label = label(pos=vector(-20,16,0), text="Acceleration: 0 m/s²", height=16, box=False, color=color.green) time_label = label(pos=vector(-20,14,0), text="Time: 0 s", height=16, box=False, color=color.blue) while head.pos.y > 1: rate(100) t += dt # Open parachute near ground if head.pos.y < 6 and not parachute_deployed: parachute_deployed = True for pt in parachute_points: pt.visible = True for rope in ropes: rope.visible = True # Drag depending on parachute drag = drag_big if parachute_deployed else drag_small # ACCELERATION a = g - drag*v*abs(v) v = v + a*dt dy = v*dt # MOVE SKYDIVER head.pos.y -= dy body.pos.y -= dy left_arm.pos.y -= dy right_arm.pos.y -= dy left_leg.pos.y -= dy right_leg.pos.y -= dy # UPDATE PARACHUTE POSITION if parachute_deployed: idx = 0 for i in range(num_segments + 1): y_offset = canopy_height * sin(pi * i / num_segments) r = canopy_radius * cos(pi * i / num_segments) num_circle_points = 30 for j in range(num_circle_points): theta = 2 * pi * j / num_circle_points x = r * cos(theta) z = r * sin(theta) parachute_points[idx].pos = vector(x, head.pos.y + y_offset + 2, z) idx += 1 # Update ropes for i, rope in enumerate(ropes): rope.pos = parachute_points[i * num_circle_points].pos rope.axis = head.pos - rope.pos # UPDATE LABELS vel_label.text = f"Velocity: {v:.2f} m/s" acc_label.text = f"Acceleration: {a:.2f} m/s²" time_label.text = f"Time: {t:.2f} s"
